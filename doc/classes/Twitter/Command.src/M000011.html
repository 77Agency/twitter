      def friends
        config = create_or_find_config
        
        begin
          puts
          Twitter::Base.new(config['email'], config['password']).friends.each do |u|
            puts "#{u.name} (#{u.screen_name}) last updated #{u.status.relative_created_at}\n-- #{u.status.text}"
            puts
          end
        rescue
          puts error_msg
        end
      end
