      def new_from_xml(xml)
        User.new do |u|
          u.id            = (xml).at('id').innerHTML
          u.name          = (xml).at('name').innerHTML
          u.screen_name   = (xml).at('screen_name').innerHTML
          u.status        = Status.new_from_xml(xml) if (xml).at('status')
        end
      end
